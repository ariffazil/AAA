# webhook_forge.py

from fastapi import FastAPI, Request, HTTPException
import os
import hmac
import hashlib
import httpx
from datetime import datetime, timezone

app = FastAPI()

WEBHOOK_SECRET = ***"WEBHOOK_SECRET"]

ARIFOS_URL = os.getenv("ARIFOS_URL", "http://arifos:8080")
VAULT999_URL = os.getenv("VAULT999_URL", "http://vault999-writer:5001")


def verify_signature(raw_body: bytes, signature: str) -> bool:
 expected = hmac.new(
 WEBHOOK_SECRET.encode("utf-8"),
 raw_body,
 hashlib.sha256
 ).hexdigest()

 return hmac.compare_digest(signature, expected)


@app.post("/webhook/forge")
async def webhook_forge(request: Request):
 raw_body = await request.body()
 signature = request.headers.get("x-arifos-signature")

 if not signature:
 raise HTTPException(status_code=401, detail="HOLD: missing signature")

 if not verify_signature(raw_body, signature):
 raise HTTPException(status_code=403, detail="VOID: invalid signature")

 event = await request.json()

 envelope = {
 "source": event.get("source", "unknown"),
 "actor": event.get("actor", "unknown"),
 "intent": event.get("intent", "unknown"),
 "target": event.get("target", "unknown"),
 "payload": event.get("payload", {}),
 "received_at": datetime.now(timezone.utc).isoformat(),
 "governance": {
 "entry": "webhook",
 "requires_f13": True,
 "authority": "APEX",
 "mode": "adjudicate_then_seal"
 }
 }

 async with httpx.AsyncClient(timeout=30) as client:
 risk = await client.post(
 f"{ARIFOS_URL}/governance/risk-classify",
 json=envelope
 )

 risk_data = risk.json()

 verdict = risk_data.get("verdict", "HOLD")

 seal_payload = {
 "event": envelope,
 "verdict": verdict,
 "risk": risk_data,
 "sealed_at": datetime.now(timezone.utc).isoformat()
 }

 sealed = await client.post(
 f"{VAULT999_URL}/seal",
 json=seal_payload
 )

 return {
 "ok": True,
 "verdict": verdict,
 "sealed": sealed.json()
 }
